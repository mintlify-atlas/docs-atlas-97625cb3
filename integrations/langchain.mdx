## Introduction

Helicone integrates seamlessly with LangChain, providing comprehensive observability for your chains, agents, tools, and retrievers. Track every step of your LangChain applications with full request/response logging, cost tracking, and performance monitoring.

<Note>
  For new projects, consider using the [AI Gateway](/gateway/overview) with LangChain for unified access to 100+ models.
</Note>

## Quick Start

Integrate Helicone with LangChain by configuring the base URL for your LLM provider.

<Tabs>
  <Tab title="JavaScript/TypeScript">
    <Steps>
      <Step title="Install dependencies">
        ```bash
        npm install @langchain/openai @langchain/core langchain
        ```
      </Step>

      <Step title="Set environment variables">
        ```bash
        HELICONE_API_KEY=sk-helicone-...
        OPENAI_API_KEY=sk-...
        ```
      </Step>

      <Step title="Configure LangChain with Helicone">
        ```typescript
        import { ChatOpenAI } from "@langchain/openai";

        const llm = new ChatOpenAI({
          modelName: "gpt-4o",
          openAIApiKey: process.env.OPENAI_API_KEY,
          configuration: {
            basePath: "https://oai.helicone.ai/v1",
            defaultHeaders: {
              "Helicone-Auth": `Bearer ${process.env.HELICONE_API_KEY}`,
            },
          },
        });

        const response = await llm.invoke("What is the meaning of life?");
        console.log(response.content);
        ```
      </Step>
    </Steps>
  </Tab>

  <Tab title="Python">
    <Steps>
      <Step title="Install dependencies">
        ```bash
        pip install langchain langchain-openai python-dotenv
        ```
      </Step>

      <Step title="Set environment variables">
        ```bash
        HELICONE_API_KEY=sk-helicone-...
        OPENAI_API_KEY=sk-...
        ```
      </Step>

      <Step title="Configure LangChain with Helicone">
        ```python
        from langchain_openai import ChatOpenAI
        from dotenv import load_dotenv
        import os

        load_dotenv()

        llm = ChatOpenAI(
            model="gpt-4o",
            openai_api_key=os.getenv("OPENAI_API_KEY"),
            openai_api_base="https://oai.helicone.ai/v1",
            default_headers={
                "Helicone-Auth": f"Bearer {os.getenv('HELICONE_API_KEY')}",
            },
        )

        response = llm.invoke("What is the meaning of life?")
        print(response.content)
        ```
      </Step>
    </Steps>
  </Tab>
</Tabs>

## Advanced Features

### Tracking Chains

Automatically track entire LangChain chains with all intermediate steps.

<Tabs>
  <Tab title="JavaScript/TypeScript">
    ```typescript
    import { ChatOpenAI } from "@langchain/openai";
    import { PromptTemplate } from "@langchain/core/prompts";
    import { StringOutputParser } from "@langchain/core/output_parsers";

    const llm = new ChatOpenAI({
      modelName: "gpt-4o",
      configuration: {
        basePath: "https://oai.helicone.ai/v1",
        defaultHeaders: {
          "Helicone-Auth": `Bearer ${process.env.HELICONE_API_KEY}`,
          "Helicone-Property-Chain": "translation-chain",
        },
      },
    });

    const prompt = PromptTemplate.fromTemplate(
      "Translate the following text to {language}: {text}"
    );

    const chain = prompt.pipe(llm).pipe(new StringOutputParser());

    const result = await chain.invoke({
      language: "French",
      text: "Hello, how are you?",
    });

    console.log(result);
    ```
  </Tab>

  <Tab title="Python">
    ```python
    from langchain_openai import ChatOpenAI
    from langchain.prompts import PromptTemplate
    from langchain.schema.output_parser import StrOutputParser
    import os

    llm = ChatOpenAI(
        model="gpt-4o",
        openai_api_base="https://oai.helicone.ai/v1",
        default_headers={
            "Helicone-Auth": f"Bearer {os.getenv('HELICONE_API_KEY')}",
            "Helicone-Property-Chain": "translation-chain",
        },
    )

    prompt = PromptTemplate.from_template(
        "Translate the following text to {language}: {text}"
    )

    chain = prompt | llm | StrOutputParser()

    result = chain.invoke({
        "language": "French",
        "text": "Hello, how are you?",
    })

    print(result)
    ```
  </Tab>
</Tabs>

### Tracking Agents

Monitor agent actions, tool calls, and reasoning steps.

<Tabs>
  <Tab title="JavaScript/TypeScript">
    ```typescript
    import { ChatOpenAI } from "@langchain/openai";
    import { AgentExecutor, createOpenAIFunctionsAgent } from "langchain/agents";
    import { DynamicTool } from "@langchain/core/tools";
    import { ChatPromptTemplate, MessagesPlaceholder } from "@langchain/core/prompts";

    const llm = new ChatOpenAI({
      modelName: "gpt-4o",
      configuration: {
        basePath: "https://oai.helicone.ai/v1",
        defaultHeaders: {
          "Helicone-Auth": `Bearer ${process.env.HELICONE_API_KEY}`,
          "Helicone-Property-Type": "agent",
        },
      },
    });

    const tools = [
      new DynamicTool({
        name: "get_weather",
        description: "Get the current weather for a location",
        func: async (location: string) => {
          return `The weather in ${location} is sunny, 72°F`;
        },
      }),
    ];

    const prompt = ChatPromptTemplate.fromMessages([
      ["system", "You are a helpful assistant."],
      ["human", "{input}"],
      new MessagesPlaceholder("agent_scratchpad"),
    ]);

    const agent = await createOpenAIFunctionsAgent({
      llm,
      tools,
      prompt,
    });

    const agentExecutor = new AgentExecutor({
      agent,
      tools,
    });

    const result = await agentExecutor.invoke({
      input: "What's the weather like in San Francisco?",
    });

    console.log(result.output);
    ```
  </Tab>

  <Tab title="Python">
    ```python
    from langchain_openai import ChatOpenAI
    from langchain.agents import AgentExecutor, create_openai_functions_agent
    from langchain.tools import Tool
    from langchain.prompts import ChatPromptTemplate, MessagesPlaceholder
    import os

    llm = ChatOpenAI(
        model="gpt-4o",
        openai_api_base="https://oai.helicone.ai/v1",
        default_headers={
            "Helicone-Auth": f"Bearer {os.getenv('HELICONE_API_KEY')}",
            "Helicone-Property-Type": "agent",
        },
    )

    def get_weather(location: str) -> str:
        return f"The weather in {location} is sunny, 72°F"

    tools = [
        Tool(
            name="get_weather",
            description="Get the current weather for a location",
            func=get_weather,
        ),
    ]

    prompt = ChatPromptTemplate.from_messages([
        ("system", "You are a helpful assistant."),
        ("human", "{input}"),
        MessagesPlaceholder("agent_scratchpad"),
    ])

    agent = create_openai_functions_agent(llm, tools, prompt)
    agent_executor = AgentExecutor(agent=agent, tools=tools)

    result = agent_executor.invoke({
        "input": "What's the weather like in San Francisco?"
    })

    print(result["output"])
    ```
  </Tab>
</Tabs>

## RAG Applications

Track retrieval-augmented generation workflows.

<Tabs>
  <Tab title="JavaScript/TypeScript">
    ```typescript
    import { ChatOpenAI } from "@langchain/openai";
    import { MemoryVectorStore } from "langchain/vectorstores/memory";
    import { OpenAIEmbeddings } from "@langchain/openai";
    import { createRetrievalChain } from "langchain/chains/retrieval";
    import { createStuffDocumentsChain } from "langchain/chains/combine_documents";
    import { ChatPromptTemplate } from "@langchain/core/prompts";

    const embeddings = new OpenAIEmbeddings({
      configuration: {
        basePath: "https://oai.helicone.ai/v1",
        defaultHeaders: {
          "Helicone-Auth": `Bearer ${process.env.HELICONE_API_KEY}`,
          "Helicone-Property-Type": "embeddings",
        },
      },
    });

    const llm = new ChatOpenAI({
      modelName: "gpt-4o",
      configuration: {
        basePath: "https://oai.helicone.ai/v1",
        defaultHeaders: {
          "Helicone-Auth": `Bearer ${process.env.HELICONE_API_KEY}`,
          "Helicone-Property-Type": "rag-generation",
        },
      },
    });

    // Create vector store and retrieval chain
    const vectorStore = await MemoryVectorStore.fromTexts(
      ["LangChain is a framework for LLM applications"],
      [{ source: "docs" }],
      embeddings
    );

    const retriever = vectorStore.asRetriever();

    const prompt = ChatPromptTemplate.fromTemplate(
      "Answer based on context: {context}\n\nQuestion: {input}"
    );

    const combineDocsChain = await createStuffDocumentsChain({
      llm,
      prompt,
    });

    const retrievalChain = await createRetrievalChain({
      retriever,
      combineDocsChain,
    });

    const result = await retrievalChain.invoke({
      input: "What is LangChain?",
    });

    console.log(result.answer);
    ```
  </Tab>

  <Tab title="Python">
    ```python
    from langchain_openai import ChatOpenAI, OpenAIEmbeddings
    from langchain.vectorstores import FAISS
    from langchain.chains import RetrievalQA
    from langchain.text_splitter import CharacterTextSplitter
    import os

    embeddings = OpenAIEmbeddings(
        openai_api_base="https://oai.helicone.ai/v1",
        default_headers={
            "Helicone-Auth": f"Bearer {os.getenv('HELICONE_API_KEY')}",
            "Helicone-Property-Type": "embeddings",
        },
    )

    llm = ChatOpenAI(
        model="gpt-4o",
        openai_api_base="https://oai.helicone.ai/v1",
        default_headers={
            "Helicone-Auth": f"Bearer {os.getenv('HELICONE_API_KEY')}",
            "Helicone-Property-Type": "rag-generation",
        },
    )

    # Create vector store
    texts = ["LangChain is a framework for LLM applications"]
    vectorstore = FAISS.from_texts(texts, embeddings)

    # Create retrieval chain
    qa_chain = RetrievalQA.from_chain_type(
        llm=llm,
        chain_type="stuff",
        retriever=vectorstore.as_retriever(),
    )

    result = qa_chain.invoke({"query": "What is LangChain?"})
    print(result["result"])
    ```

</Tabs>
