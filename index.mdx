    <Step title="Get your API key">
      Sign up at [helicone.ai](https://helicone.ai/signup) and get your API key from the dashboard.
    </Step>
    
    <Step title="Update your code">
      Change the `baseURL` to point to Helicone's AI Gateway:
      
      ```typescript
      import OpenAI from "openai";

      const client = new OpenAI({
        baseURL: "https://ai-gateway.helicone.ai",
        apiKey: process.env.HELICONE_API_KEY,
      });

      const response = await client.chat.completions.create({
        model: "gpt-4o-mini",
        messages: [{ role: "user", content: "Hello!" }]
      });
      ```
    </Step>
    
    <Step title="View your logs">
      Check your requests, traces, and analytics in the [Helicone dashboard](https://us.helicone.ai/dashboard).
    </Step>
  </Steps>
